<script>

function formatTime(date) {

  return date.toLocaleTimeString([], {

    hour: 'numeric',

    minute: '2-digit',

    hour12: true

  });

}


function calculateEndTime() {

  const startTime = document.getElementById("startTime").value;

  const totalMinutes = parseInt(document.getElementById("totalMinutes").value);

  const approvedBreaks = parseInt(document.getElementById("approvedBreaks").value);

  const breaksTaken = parseInt(document.getElementById("breaksTaken").value);


  if (!startTime || isNaN(totalMinutes) || isNaN(approvedBreaks) || isNaN(breaksTaken)) {

    document.getElementById("output").innerText = "Please fill out all fields.";

    document.getElementById("notesOutput").innerText = "";

    return;

  }


  if (breaksTaken > approvedBreaks) {

    document.getElementById("output").innerText =

      "Break minutes taken cannot exceed approved break minutes.";

    document.getElementById("notesOutput").innerText = "";

    return;

  }


  const [hours, minutes] = startTime.split(":").map(Number);

  const startDate = new Date();

  startDate.setHours(hours, minutes, 0, 0);


  const scheduledEnd = new Date(startDate.getTime() + totalMinutes * 60000);

  const unusedBreaks = approvedBreaks - breaksTaken;

  const adjustedEnd = new Date(scheduledEnd.getTime() - unusedBreaks * 60000);


  document.getElementById("output").innerHTML = `

    <strong>Exam Start Time:</strong> ${formatTime(startDate)}<br><br>


    <strong>Scheduled End Time (maximum, with all breaks):</strong><br>

    ${formatTime(scheduledEnd)}<br><br>


    <strong>Break Minutes Taken:</strong> ${breaksTaken}<br>

    <strong>Unused Break Minutes:</strong> ${unusedBreaks}<br><br>


    <strong>Adjusted End Time (enter in RegisterBlast):</strong><br>

    ${formatTime(adjustedEnd)}

  `;


  document.getElementById("notesOutput").innerText =

`RegisterBlast Note:

Student approved for ${approvedBreaks} minutes of breaks.

Student took ${breaksTaken} minutes of breaks.

Unused break time (${unusedBreaks} minutes) was removed from the scheduled exam time.

Exam end time adjusted accordingly.`;

}

</script>
